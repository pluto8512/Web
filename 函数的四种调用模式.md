 #函数的四种调用模式(非常非常重要)
1. 调用模式

> 函数模式
> 方法模式
> 构造器模式
> 上下文模式

2. 函数的定义方式

> 声明式
> 表达式式
> Function

```javascript
// 1. 函数模式
function foo () {
   console.log("1");
}
foo();
//this 表示全局对象
//任何自调用都是函数调用模式

// 2. 方法调用模式
// -> 区分
// 方法本生就是函数,但是是通过对象引导调用
// 对象.方法( 参数 )
// this 表示 引导放的对象
function foo () {
   console.log(this);
}
var o = {name:"jack"];
o.foo = foo;

foo();
o.foo();

function func () {
   console.log("2"); 
}
func.foo = foo;
func.foo(); //打印函数体

var arr = [1,2,3];
arr.foo = foo; 
arr.foo(); //打印数组对象

var arr = [1,2,3,function(){
   console.log(this);
}] ;
arr[3]();// 打印数组

// 3. 构造器模式(或者叫做构造函数模式,构造方法模式)
// 构造器的执行步骤
var p = new Person ();
// new 是一个运算符,专门用来申请创建对象
// 创建出来的对象传递给构造函数的this,利用构造函数来进行初始化
// 构造函数不写return默认返回this.

//构造函数的return语句

// 4. 上下文调用模式(环境调用模式)-> 在不同环境下不同调用模式 
// 语法区分
// call-> 函数名.call(...);
// apply -> 函数名.apply(...)
// 这两种形式功能完全一样,只是参数形式不同
// 上下文-->环境
// 用法:
// 存在上下文调用的目的: 实现借用方法
 function foo() {
   consol.log (this);
 }
 var o = { name: "jim"};

 foo();

 o.func = foo;
 o.func();
// 如果需要让函数以函数的形式调用, 可以使用
foo.apply(null)或foo.apply()

//如果希望他是以方法形式调用,可以使用
foo.apply(arg);,必须传入一个对象,对象是什么,上下文就是什么

function foo() {
   consol.log (this);
 }
var o = { name: "jim"};

foo.apply();//window
foo.apply ();//window
foo.apply (o);//object

// 练习 
function fn() {
   console.log(this);
}
var o = {
   name : "o"
};
fn.apply();// 打印window
fn.apply(o);// 借调打印o

//当有参数的时候怎么进行上下文调用
function foo (num1 , num2 ){
   console.log(this);
   return num1 + num2;
}

//函数调用模式
var res1 = foo(123, 567);

//方法调用模式
var o = {name : "jim"};
o.func = foo;
var res2 = o.func(123, 567);

//使用apply 进行调用.如果函数带有参数,apply 的第一个参数要么是null 要么是对象
// 函数模式 foo(123, 567);-->foo.apply(null,[123, 567])
// 方法模式 foo.apply(o, [123,567])
foo.apply(null,[123,567]);
foo.apply(o,[123,567]);
```
